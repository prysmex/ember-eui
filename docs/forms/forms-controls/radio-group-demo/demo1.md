```hbs template
<EuiRadioGroup
  @options={{this.radios}}
  @idSelected={{this.selectedRadioId}}
  @onChange={{set this.selectedRadioId}}
/>
<EuiSpacer />
<EuiRadioGroup
  @legend="With legend"
  @options={{this.radios}}
  @idSelected={{this.selectedRadioId}}

/>

Selected:
{{this.selectedRadioId}}
```
